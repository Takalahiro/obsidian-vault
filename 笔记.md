## 先快速回答你的两个疑问

### 1. "vault 是什么？"

**Vault = Obsidian 里的"笔记仓库"，本质就是一个文件夹**。

你在 Obsidian 里打开的那个文件夹（截图里能看到 `计算机/`、`INFO1111/`、`MATH/` 那个根目录），就叫 vault。

```
你 Obsidian 里看到的：              文件系统里实际是：
┌─────────────────┐                ┌──────────────────────────┐
│ ▶ 计算机         │                │ /Users/你/MyNotes/        │
│ ▶ DATA1001      │     就是       │  ├── 计算机/              │
│ ▶ INFO1110      │     ━━━>       │  ├── DATA1001/            │
│ ▶ MATH          │                │  ├── INFO1110/            │
└─────────────────┘                │  ├── MATH/                │
                                   │  └── .obsidian/  ← 配置  │
                                   └──────────────────────────┘
                                          ↑ 这整个就叫 vault
```

后面我让你"复制 vault 到项目目录"，意思就是：**把你 Obsidian 打开的那个根文件夹整个复制一份到项目里**。

### 2. 后期写笔记怎么快速同步？

我给你**三种方案**，按推荐度排序，后面 README 里也会再说：

```
方案 A：Git 同步（推荐）⭐⭐⭐⭐⭐
  在 Obsidian 装 "Obsidian Git" 插件
  写完笔记按 Ctrl+Shift+P → "Commit and Push"
  Cloudflare Pages 自动检测 → 重新构建（~3 分钟上线）
  
  优点：版本控制、历史可回溯、免费
  缺点：每次有 3 分钟构建延迟

方案 B：增量同步（进阶，v2 实现）⭐⭐⭐
  做一个 webhook，只重新 embed 修改过的笔记
  构建时间从 3 分钟 → 30 秒
  
方案 C：双 vault（你的原 vault + 项目 vault）⭐⭐
  原 vault 你自己用，定期 rsync 到项目 vault
  最简单但不优雅
```

**我推荐方案 A**，今天的代码就内置这个流程，README 里写好步骤。

## 第 7 步：⭐ 核心构建脚本 `scripts/build-index.mjs`

这是**最关键的脚本**，针对你的目录结构（中英文混合、文件夹层级、Obsidian 双链、相对路径图片）专门定制。

```javascript
// scripts/build-index.mjs
// 扫描 vault → 生成 index.json / tree.json / chunks.json / notes/*.md

import fs from 'node:fs/promises';
import path from 'node:path';
import { fileURLToPath } from 'node:url';
import matter from 'gray-matter';

const __dirname = path.dirname(fileURLToPath(import.meta.url));
const VAULT_DIR = path.resolve(__dirname, '../vault');
const PUBLIC_DIR = path.resolve(__dirname, '../public');
const NOTES_OUT = path.join(PUBLIC_DIR, 'notes');
const ASSETS_OUT = path.join(PUBLIC_DIR, 'assets');

// ──────────────────────────────────────────────────────────
// 工具函数
// ──────────────────────────────────────────────────────────

const SKIP_DIRS = new Set(['.obsidian', '.trash', '.git', 'node_modules']);
const SKIP_EXTS = new Set(['.canvas', '.base']); // v1 跳过 canvas/base
const IMG_EXTS = new Set(['.png', '.jpg', '.jpeg', '.gif', '.webp', '.svg', '.bmp']);
const MD_EXTS = new Set(['.md', '.markdown']);

/** 把路径转成 URL slug（保留中文，只编码空格和特殊字符）*/
function toSlug(relPath) {
  return relPath
    .replace(/\\/g, '/')            // Windows 路径
    .replace(/\.md$/, '')           // 去 .md
    .split('/')
    .map(seg => encodeURIComponent(seg).replace(/%20/g, ' ')) // 保留空格可读
    .join('/');
}

/** 把笔记标题统一（用文件名去掉扩展名）*/
function noteTitle(relPath, frontmatter) {
  if (frontmatter?.title) return frontmatter.title;
  return path.basename(relPath).replace(/\.md$/, '');
}

/** 递归扫描目录 */
async function walk(dir, base = dir) {
  const entries = await fs.readdir(dir, { withFileTypes: true });
  const result = [];
  for (const entry of entries) {
    if (SKIP_DIRS.has(entry.name)) continue;
    if (entry.name.startsWith('.')) continue;  // 隐藏文件
    
    const full = path.join(dir, entry.name);
    const rel = path.relative(base, full);
    
    if (entry.isDirectory()) {
      const children = await walk(full, base);
      result.push({ type: 'dir', name: entry.name, path: rel, children });
    } else {
      const ext = path.extname(entry.name).toLowerCase();
      if (SKIP_EXTS.has(ext)) continue;
      result.push({ type: 'file', name: entry.name, path: rel, ext });
    }
  }
  return result;
}

/** 把扫描结果转成树（保留层级，去掉空目录）*/
function buildTree(entries) {
  return entries
    .map(e => {
      if (e.type === 'dir') {
        const children = buildTree(e.children);
        if (children.length === 0) return null;
        return { type: 'dir', name: e.name, path: e.path, children };
      }
      if (MD_EXTS.has(e.ext)) {
        return { 
          type: 'note', 
          name: e.name.replace(/\.md$/, ''), 
          path: e.path,
          slug: toSlug(e.path)
        };
      }
      return null;
    })
    .filter(Boolean);
}

/** 收集所有 md 文件（扁平列表）*/
function collectNotes(entries, acc = []) {
  for (const e of entries) {
    if (e.type === 'dir') collectNotes(e.children, acc);
    else if (MD_EXTS.has(e.ext)) acc.push(e);
  }
  return acc;
}

/** 收集所有图片文件 */
function collectImages(entries, acc = []) {
  for (const e of entries) {
    if (e.type === 'dir') collectImages(e.children, acc);
    else if (IMG_EXTS.has(e.ext)) acc.push(e);
  }
  return acc;
}

// ──────────────────────────────────────────────────────────
// 双链解析
// ──────────────────────────────────────────────────────────

/** 解析 [[xxx]] / [[xxx|别名]] / [[xxx#标题]] / [[xxx#标题|别名]] */
function parseWikiLinks(content) {
  const links = [];
  const regex = /\[\[([^\]]+)\]\]/g;
  let match;
  while ((match = regex.exec(content)) !== null) {
    const raw = match[1];
    // 跳过图片嵌入 ![[xxx.png]]（前面有 !）
    if (content[match.index - 1] === '!') continue;
    
    let target = raw, heading = null, alias = null;
    if (raw.includes('|')) {
      [target, alias] = raw.split('|');
    }
    if (target.includes('#')) {
      [target, heading] = target.split('#');
    }
    links.push({ target: target.trim(), heading, alias, raw });
  }
  return links;
}

/** 解析 #tag（支持中文标签）*/
function parseTags(content) {
  const tags = new Set();
  const regex = /(?:^|\s)#([\p{L}\p{N}_/-]+)/gu;
  let match;
  while ((match = regex.exec(content)) !== null) {
    tags.add(match[1]);
  }
  return [...tags];
}

/** 提取标题树（H1-H4）*/
function parseHeadings(content) {
  const headings = [];
  const lines = content.split('\n');
  let inCodeBlock = false;
  for (const line of lines) {
    if (line.startsWith('```')) { inCodeBlock = !inCodeBlock; continue; }
    if (inCodeBlock) continue;
    const m = line.match(/^(#{1,6})\s+(.+?)\s*$/);
    if (m) {
      headings.push({ level: m[1].length, text: m[2].trim() });
    }
  }
  return headings;
}

// ──────────────────────────────────────────────────────────
// 双链消歧（同目录 > 子目录 > 全局）
// ──────────────────────────────────────────────────────────

function resolveLinks(allNotes) {
  // 建立索引：title -> [notePath, ...]
  const titleIndex = new Map();
  for (const note of allNotes) {
    const title = noteTitle(note.relPath, note.frontmatter);
    if (!titleIndex.has(title)) titleIndex.set(title, []);
    titleIndex.get(title).push(note.relPath);
  }
  
  // 为每个 note 解析双链
  for (const note of allNotes) {
    note.resolvedLinks = [];
    for (const link of note.rawLinks) {
      // 如果 link.target 含 /，说明是带路径的链接，直接匹配
      if (link.target.includes('/')) {
        const matched = allNotes.find(n => 
          n.relPath.replace(/\.md$/, '') === link.target ||
          n.relPath === link.target + '.md'
        );
        note.resolvedLinks.push({
          ...link,
          resolved: matched ? matched.relPath : null,
          ambiguous: false
        });
        continue;
      }
      
      const candidates = titleIndex.get(link.target) || [];
      if (candidates.length === 0) {
        note.resolvedLinks.push({ ...link, resolved: null, ambiguous: false });
      } else if (candidates.length === 1) {
        note.resolvedLinks.push({ ...link, resolved: candidates[0], ambiguous: false });
      } else {
        // 多个候选：同目录优先
        const noteDir = path.dirname(note.relPath);
        const sameDir = candidates.find(c => path.dirname(c) === noteDir);
        if (sameDir) {
          note.resolvedLinks.push({ ...link, resolved: sameDir, ambiguous: false });
        } else {
          // 子目录优先
          const subDir = candidates.find(c => c.startsWith(noteDir + '/'));
          if (subDir) {
            note.resolvedLinks.push({ ...link, resolved: subDir, ambiguous: false });
          } else {
            // 取第一个并标记歧义
            console.warn(`⚠️  双链歧义 "${link.target}" in ${note.relPath}：候选 ${candidates.join(', ')}`);
            note.resolvedLinks.push({ ...link, resolved: candidates[0], ambiguous: true, candidates });
          }
        }
      }
    }
  }
}

// ──────────────────────────────────────────────────────────
// 图片路径重写
// ──────────────────────────────────────────────────────────

function rewriteImagePaths(content, noteRelPath, imageIndex) {
  const noteDir = path.dirname(noteRelPath);
  
  // Obsidian 嵌入 ![[xxx.png]] 或 ![[xxx.png|宽度]]
  content = content.replace(/!\[\[([^\]]+)\]\]/g, (match, inner) => {
    let [imgRef, size] = inner.split('|');
    imgRef = imgRef.trim();
    const ext = path.extname(imgRef).toLowerCase();
    if (!IMG_EXTS.has(ext)) return match; // 非图片，可能是笔记嵌入
    
    // 在 imageIndex 里找
    const resolved = findImage(imgRef, noteDir, imageIndex);
    if (!resolved) {
      console.warn(`⚠️  图片找不到: ${imgRef} (in ${noteRelPath})`);
      return match;
    }
    const sizeAttr = size ? ` width="${size}"` : '';
    return `<img src="/assets/${encodeURI(resolved)}" alt="${path.basename(imgRef)}"${sizeAttr} />`;
  });
  
  // 标准 markdown ![alt](path)
  content = content.replace(/!\[([^\]]*)\]\(([^)]+)\)/g, (match, alt, src) => {
    src = src.trim();
    if (src.startsWith('http://') || src.startsWith('https://') || src.startsWith('/')) return match;
    
    // 相对路径解析
    const resolved = path.posix.normalize(path.posix.join(noteDir, src));
    return `![${alt}](/assets/${encodeURI(resolved)})`;
  });
  
  return content;
}

function findImage(imgRef, noteDir, imageIndex) {
  // 1. 同目录优先
  const sameDir = path.posix.join(noteDir, imgRef);
  if (imageIndex.has(sameDir)) return sameDir;
  
  // 2. picture 子目录
  const picSub = path.posix.join(noteDir, 'picture', path.basename(imgRef));
  if (imageIndex.has(picSub)) return picSub;
  
  // 3. 全局按文件名匹配
  const basename = path.basename(imgRef);
  for (const img of imageIndex.keys()) {
    if (path.basename(img) === basename) return img;
  }
  
  return null;
}

// ──────────────────────────────────────────────────────────
// 切块（按 H2，超长再细分）
// ──────────────────────────────────────────────────────────

const MAX_CHUNK_CHARS = 1500;
const MIN_CHUNK_CHARS = 100;

function chunkByHeadings(content, noteMeta) {
  // 去掉 frontmatter（已经在外面 parse 过了）
  const body = content;
  const lines = body.split('\n');
  
  const chunks = [];
  let currentH1 = '', currentH2 = '', currentH3 = '';
  let buffer = [];
  let bufferStart = { h1: '', h2: '', h3: '' };
  let inCodeBlock = false;
  
  const flush = () => {
    const text = buffer.join('\n').trim();
    if (text.length >= MIN_CHUNK_CHARS) {
      if (text.length <= MAX_CHUNK_CHARS) {
        chunks.push({
          ...noteMeta,
          h1: bufferStart.h1,
          h2: bufferStart.h2,
          h3: bufferStart.h3,
          text,
          chars: text.length
        });
      } else {
        // 超长：按段落滑窗切
        const paragraphs = text.split(/\n\n+/);
        let sub = '';
        for (const p of paragraphs) {
          if ((sub + '\n\n' + p).length > MAX_CHUNK_CHARS && sub) {
            chunks.push({
              ...noteMeta,
              h1: bufferStart.h1,
              h2: bufferStart.h2,
              h3: bufferStart.h3,
              text: sub.trim(),
              chars: sub.length
            });
            sub = p;
          } else {
            sub = sub ? sub + '\n\n' + p : p;
          }
        }
        if (sub.trim().length >= MIN_CHUNK_CHARS) {
          chunks.push({
            ...noteMeta,
            h1: bufferStart.h1,
            h2: bufferStart.h2,
            h3: bufferStart.h3,
            text: sub.trim(),
            chars: sub.length
          });
        }
      }
    }
    buffer = [];
  };
  
  for (const line of lines) {
    if (line.startsWith('```')) inCodeBlock = !inCodeBlock;
    
    if (!inCodeBlock) {
      const m = line.match(/^(#{1,3})\s+(.+?)\s*$/);
      if (m) {
        flush();
        const level = m[1].length;
        const text = m[2].trim();
        if (level === 1) { currentH1 = text; currentH2 = ''; currentH3 = ''; }
        else if (level === 2) { currentH2 = text; currentH3 = ''; }
        else if (level === 3) { currentH3 = text; }
        bufferStart = { h1: currentH1, h2: currentH2, h3: currentH3 };
        buffer.push(line);
        continue;
      }
    }
    buffer.push(line);
  }
  flush();
  
  return chunks;
}

// 主流程
// ──────────────────────────────────────────────────────────

async function ensureDir(dir) {
  await fs.mkdir(dir, { recursive: true });
}

async function copyAsset(src, dest) {
  await ensureDir(path.dirname(dest));
  await fs.copyFile(src, dest);
}

async function main() {
  console.log('📂 扫描 vault:', VAULT_DIR);
  const entries = await walk(VAULT_DIR);
  
  // 构建树
  const tree = buildTree(entries);
  
  // 收集笔记和图片
  const noteFiles = collectNotes(entries);
  const imageFiles = collectImages(entries);
  console.log(`  发现 ${noteFiles.length} 篇笔记, ${imageFiles.length} 张图片`);
  
  // 建立图片索引（用 posix 路径，便于跨平台）
  const imageIndex = new Map();
  for (const img of imageFiles) {
    imageIndex.set(img.path.replace(/\\/g, '/'), img);
  }
  
  // 准备输出目录
  await ensureDir(NOTES_OUT);
  await ensureDir(ASSETS_OUT);
  
  // 复制图片
  console.log('🖼  复制图片...');
  for (const img of imageFiles) {
    const src = path.join(VAULT_DIR, img.path);
    const dest = path.join(ASSETS_OUT, img.path);
    await copyAsset(src, dest);
  }
  
  // 处理每篇笔记
  console.log('📝 处理笔记...');
  const allNotes = [];
  
  for (const file of noteFiles) {
    const full = path.join(VAULT_DIR, file.path);
    const raw = await fs.readFile(full, 'utf-8');
    const { data: frontmatter, content } = matter(raw);
    
    const headings = parseHeadings(content);
    const rawLinks = parseWikiLinks(content);
    const tags = [
      ...(frontmatter.tags || []),
      ...parseTags(content)
    ];
    
    const title = noteTitle(file.path, frontmatter);
    const slug = toSlug(file.path);
    
    allNotes.push({
      relPath: file.path.replace(/\\/g, '/'),
      slug,
      title,
      frontmatter,
      content,
      headings,
      rawLinks,
      tags: [...new Set(tags)],
      chars: content.length
    });
  }
  
  // 双链消歧
  console.log('🔗 解析双链...');
  resolveLinks(allNotes);
  
  // 反链索引
  const backlinks = new Map();  // relPath -> [{from, raw}]
  for (const note of allNotes) {
    for (const link of note.resolvedLinks) {
      if (!link.resolved) continue;
      if (!backlinks.has(link.resolved)) backlinks.set(link.resolved, []);
      backlinks.get(link.resolved).push({ from: note.relPath, raw: link.raw });
    }
  }
  
  // 重写图片路径并写出每篇笔记
  console.log('🖼  重写图片路径并写出笔记...');
  for (const note of allNotes) {
    const rewritten = rewriteImagePaths(note.content, note.relPath, imageIndex);
    const outPath = path.join(NOTES_OUT, note.relPath);
    await ensureDir(path.dirname(outPath));
    await fs.writeFile(outPath, rewritten, 'utf-8');
    note.contentRewritten = rewritten;
  }
  
  // 切块
  console.log('✂️  切块（为 embedding 准备）...');
  const allChunks = [];
  for (const note of allNotes) {
    const meta = {
      notePath: note.relPath,
      slug: note.slug,
      title: note.title
    };
    const chunks = chunkByHeadings(note.contentRewritten, meta);
    chunks.forEach((c, i) => {
      c.id = `${note.slug}::${i}`;
      c.index = i;
    });
    allChunks.push(...chunks);
  }
  console.log(`  共生成 ${allChunks.length} 个 chunks`);
  
  // 生成 index.json（笔记元数据）
  const index = allNotes.map(n => ({
    path: n.relPath,
    slug: n.slug,
    title: n.title,
    frontmatter: n.frontmatter,
    headings: n.headings,
    tags: n.tags,
    links: n.resolvedLinks
      .filter(l => l.resolved)
      .map(l => ({ to: l.resolved, heading: l.heading, alias: l.alias })),
    backlinks: (backlinks.get(n.relPath) || []).map(b => b.from),
    chars: n.chars
  }));
  
  // 写出
  await fs.writeFile(
    path.join(PUBLIC_DIR, 'index.json'),

  );
  await fs.writeFile(
